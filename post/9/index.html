 cout<<"请输入你猜测的数据：a = ";

#include<iostream>

    当然在这个地方我们使用的是无限循环跳出的方法，另外的话我们可以使用另一种方法——满足条件循环。
  • 源代码：
#include<iostream>
using namespace std;
int main()
{
	bool i =false;
	int j=1;
	while(i == false)
	{
		int a,n=2;
		if(j == 1)
		{
			cout<<"请猜一猜我心里在想什么！猜对有奖励的噢！"<<endl;
		}
		cout<<"请输入你猜测的数据：a = "; 
		cin>>a;
		if(a == n)
		{
			cout<<"恭喜你猜对了！奖励你CSP初赛模拟题一套！"<<endl;
			i = true;
		}
		else
		{
			cout<<"抱歉啊，猜错了。给你个安慰奖：CSP初赛模拟题两套！\n"<<endl;
			cout<<"请再猜一猜吧："<<endl; 
		}
		j++;
	}
	return 0;
}
